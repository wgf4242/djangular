## 5 - A Django URL Catch-all.mp4

    # settings.py TEMPLATES
    'DIRS': [os.path.join(BASE_DIR, 'templates')],
    STATIC_ROOT = os.path.join(os.path.dirname(BASE_DIR), 'static-root')
    STATICFILES_DIRS = [
        os.path.join(BASE_DIR, 'static')
    ]
